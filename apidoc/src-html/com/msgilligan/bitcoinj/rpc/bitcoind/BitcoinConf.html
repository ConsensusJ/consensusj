001package com.msgilligan.bitcoinj.rpc.bitcoind;
002
003import com.msgilligan.bitcoinj.rpc.RPCConfig;
004import org.bitcoinj.params.MainNetParams;
005
006import java.net.URI;
007import java.net.URISyntaxException;
008import java.util.HashMap;
009
010/**
011 * Bitcoin configuration from bitcoin.conf
012 * First cut is just a map, but future versions may be strongly typed
013 */
014public class BitcoinConf extends HashMap<String, String> {
015
016    public RPCConfig getRPCConfig() {
017        URI uri = null;
018        try {
019            uri = new URI("http://" + get("rpcconnect") + ":" + get("rpcport"));
020        } catch (URISyntaxException e) {
021            try {
022                uri = new URI("http://127.0.0.1:8332");
023            } catch (URISyntaxException e1) {
024                throw new RuntimeException("Error creating RPC URI", e1);
025            }
026        }
027        // TODO: Determine MainNet, TestNet, or RegTest from contents of .conf file
028        RPCConfig config = new RPCConfig(MainNetParams.get(), uri,
029                get("rpcuser"), get("rpcpassword"));
030        return config;
031    }
032}
