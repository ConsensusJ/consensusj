001package com.msgilligan.bitcoinj.rpc;
002
003import java.net.URI;
004import java.net.URISyntaxException;
005
006/**
007 * Utility class with default connection information for Bitcoin JSON-RPC
008 */
009public class RPCURI {
010    public static final String rpcproto = "http";
011    public static final String rpcssl = "https";
012    public static final String rpchost = "127.0.0.1";
013    public static final String rpcfile = "/";
014
015    public static final int RPCPORT_MAINNET = 8332;
016    public static final int RPCPORT_TESTNET = 18332;
017    public static final int RPCPORT_REGTEST = 18332;
018
019    public static URI getDefaultMainNetURI() {
020        try {
021            return new URI(rpcproto, null, rpchost, RPCPORT_MAINNET, rpcfile, null, null);
022        } catch (URISyntaxException e) {
023            return null;
024        }
025    }
026
027    public static URI getDefaultTestNetURI() {
028        try {
029            return new URI(rpcproto, null, rpchost, RPCPORT_TESTNET, rpcfile, null, null);
030        } catch (URISyntaxException e) {
031            return null;
032        }
033    }
034
035    public static URI getDefaultRegTestURI() {
036        try {
037            return new URI(rpcproto, null, rpchost, RPCPORT_REGTEST, rpcfile, null, null);
038        } catch (URISyntaxException e) {
039            return null;
040        }
041    }
042
043}
