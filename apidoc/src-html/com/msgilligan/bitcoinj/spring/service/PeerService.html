016package com.msgilligan.bitcoinj.spring.service;
017
018import org.bitcoinj.core.listeners.OnTransactionBroadcastListener;
019import org.bitcoinj.net.discovery.PeerDiscovery;
020import org.springframework.messaging.simp.SimpMessageSendingOperations;
021
022import org.bitcoinj.core.*;
023
024import javax.annotation.PostConstruct;
025import javax.inject.Inject;
026import javax.inject.Named;
027import java.security.Principal;
028import java.util.List;
029
030/**
031 * Extend PeerGroupService with additional HTTP and WebSocket/STOMP services
032 * for Peers and Transactions
033 */
034@Named
035public class PeerService extends PeerGroupService {
036    private final SimpMessageSendingOperations messagingTemplate;
037
038    @Inject
039    public PeerService(NetworkParameters params,
040                       PeerDiscovery peerDiscovery,
041                       SimpMessageSendingOperations messagingTemplate) {
042        super(params, peerDiscovery);
043        this.messagingTemplate = messagingTemplate;
044    }
045
046    @PostConstruct
047    @Override
048    public void start() {
049        super.start();
050        peerGroup.addOnTransactionBroadcastListener(new MyPeerEventListener() );
051    }
052
053
054    public List<Peer> getPeers() {
055        List<Peer> peers = peerGroup.getConnectedPeers();
056        return peers;
057    }
058
059    public void listPeers(Principal principal) {
060        List<Peer> peers = peerGroup.getConnectedPeers();
061        this.messagingTemplate.convertAndSendToUser(principal.getName(), "/queue/peers", peers);
062    }
063
064    void onPGTransaction(Peer peer, Transaction tx) {
065        this.messagingTemplate.convertAndSend("/topic/tx", tx);
066    }
067
068    private class MyPeerEventListener implements OnTransactionBroadcastListener {
069        @Override
070        public void onTransaction(Peer peer, Transaction tx) {
071            System.out.println("Got transaction: " + tx.getHashAsString());
072            onPGTransaction(peer, tx);
073        }
074    }
075}
