001package com.msgilligan.bitcoinj.cli;
002
003import org.consensusj.jsonrpc.JsonRPCException;
004
005import java.io.IOException;
006
007/**
008 * A test CLI client that simply gets the blockCount
009 */
010public class GetBlockCount extends CliCommand {
011    public final static String commandName = "getblockcount";
012
013    public GetBlockCount(String[] args) {
014        super(commandName, new CliOptions(), args);
015    }
016
017    public static void main(String[] args) throws Exception {
018        GetBlockCount command = new GetBlockCount(args);
019        Integer status = command.run();
020        System.exit(status);
021    }
022
023    @Override
024    public Integer runImpl() throws IOException {
025        Integer blockCount = -1;
026        try {
027            blockCount = client.getBlockCount();
028        } catch (JsonRPCException e) {
029            e.printStackTrace();
030            return 1;
031        } catch (IOException e) {
032            e.printStackTrace();
033            return 1;
034        }
035        pwout.println("Block count: " + blockCount);
036        return 0;
037    }
038}
