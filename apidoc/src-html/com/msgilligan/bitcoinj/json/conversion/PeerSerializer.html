001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonSerializer;
006import com.fasterxml.jackson.databind.SerializerProvider;
007import org.bitcoinj.core.Peer;
008
009import java.io.IOException;
010
011/**
012 * Custom Serialization of bitcoinj Peer class
013 */
014public class PeerSerializer extends JsonSerializer<Peer> {
015    @Override
016    public void serialize(Peer value, JsonGenerator jgen, SerializerProvider provider)
017            throws IOException, JsonProcessingException {
018        jgen.writeStartObject();
019        jgen.writeStringField("socketAddress", value.getAddress().toSocketAddress().toString());
020        jgen.writeNumberField("remoteVersion", value.getPeerVersionMessage().clientVersion);
021        jgen.writeNumberField("bestHeight", value.getPeerVersionMessage().bestHeight);
022        jgen.writeEndObject();
023    }
024}
