001package com.msgilligan.bitcoinj.json.conversion;
002
003/**
004 * Hex string to hex conversion utility class.
005 */
006public class HexUtil {
007    /**
008     * Converts a hex-encoded string into a byte array.
009     *
010     * @param s A string to convert
011     * @return The byte array
012     */
013    public static byte[] hexStringToByteArray(String s) {
014        int len = s.length();
015        byte[] data = new byte[len / 2];
016        for (int i = 0; i < len; i += 2) {
017            data[i / 2] = (byte) ((Character.digit(s.charAt(i), 16) << 4)
018                    + Character.digit(s.charAt(i + 1), 16));
019        }
020        return data;
021    }
022}
