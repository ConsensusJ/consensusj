001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Sha256Hash;
007
008/**
009 *
010 */
011@JsonIgnoreProperties(ignoreUnknown = true)
012public class ChainTip {
013    private final int height;
014    private final Sha256Hash hash;
015    private final int branchlen;
016    private final String status;
017
018    @JsonCreator
019    public ChainTip(@JsonProperty("height")     int height,
020                    @JsonProperty("hash")       Sha256Hash hash,
021                    @JsonProperty("branchlen")  int branchlen,
022                    @JsonProperty("status")     String status) {
023        this.height = height;
024        this.hash = hash;
025        this.branchlen = branchlen;
026        this.status = status;
027    }
028
029    public int getHeight() {
030        return height;
031    }
032
033    public Sha256Hash getHash() {
034        return hash;
035    }
036
037    public int getBranchlen() {
038        return branchlen;
039    }
040
041    public String getStatus() {
042        return status;
043    }
044}
