001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Coin;
007
008import java.math.BigDecimal;
009
010/**
011 * POJO for `getinfo` RPC response.
012 */
013@JsonIgnoreProperties(ignoreUnknown = true)
014public class ServerInfo {
015    private final int version;
016    private final int protocolversion;
017    private final int walletversion;
018    private final Coin balance;
019    private final int blocks;
020    private final int timeoffset;
021    private final int connections;
022    private final String proxy;
023    private final BigDecimal difficulty;
024    private final Boolean testnet;
025    private final int keypoololdest;
026    private final int keypoolsize;
027    private final Coin paytxfee;
028    private final Coin relayfee;
029    private final String errors;
030
031    @JsonCreator
032    public ServerInfo(@JsonProperty("version")          int version,
033                      @JsonProperty("protocolversion")  int protocolversion,
034                      @JsonProperty("walletversion")    int walletversion,
035                      @JsonProperty("balance")          Coin balance,
036                      @JsonProperty("blocks")           int blocks,
037                      @JsonProperty("timeoffset")       int timeoffset,
038                      @JsonProperty("connections")      int connections,
039                      @JsonProperty("proxy")            String proxy,
040                      @JsonProperty("difficulty")       BigDecimal difficulty,
041                      @JsonProperty("testnet")          Boolean testnet,
042                      @JsonProperty("keypoololdest")    int keypoololdest,
043                      @JsonProperty("keypoolsize")      int keypoolsize,
044                      @JsonProperty("paytxfee")         Coin paytxfee,
045                      @JsonProperty("relayfee")         Coin relayfee,
046                      @JsonProperty("errors")           String errors) {
047        this.version = version;
048        this.protocolversion = protocolversion;
049        this.walletversion = walletversion;
050        this.balance = balance;
051        this.blocks = blocks;
052        this.timeoffset = timeoffset;
053        this.connections = connections;
054        this.proxy = proxy;
055        this.difficulty = difficulty;
056        this.testnet = testnet;
057        this.keypoololdest = keypoololdest;
058        this.keypoolsize = keypoolsize;
059        this.paytxfee = paytxfee;
060        this.relayfee = relayfee;
061        this.errors = errors;
062    }
063
064    public int getVersion() {
065        return version;
066    }
067
068    public int getProtocolversion() {
069        return protocolversion;
070    }
071
072    public int getWalletversion() {
073        return walletversion;
074    }
075
076    public Coin getBalance() {
077        return balance;
078    }
079
080    public int getBlocks() {
081        return blocks;
082    }
083
084    public int getTimeoffset() {
085        return timeoffset;
086    }
087
088    public int getConnections() {
089        return connections;
090    }
091
092    public String getProxy() {
093        return proxy;
094    }
095
096    public BigDecimal getDifficulty() {
097        return difficulty;
098    }
099
100    public Boolean getTestnet() {
101        return testnet;
102    }
103
104    public int getKeypoololdest() {
105        return keypoololdest;
106    }
107
108    public int getKeypoolsize() {
109        return keypoolsize;
110    }
111
112    public Coin getPaytxfee() {
113        return paytxfee;
114    }
115
116    public Coin getRelayfee() {
117        return relayfee;
118    }
119
120    public String getErrors() {
121        return errors;
122    }
123}
