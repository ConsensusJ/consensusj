001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006
007/**
008 *
009 */
010@JsonIgnoreProperties(ignoreUnknown = true)
011public class SignedRawTransaction {
012    private final String hex;
013    private final boolean complete;
014
015    @JsonCreator
016    public SignedRawTransaction(@JsonProperty("hex")        String   hex,
017                                @JsonProperty("complete")   boolean  complete) {
018        this.hex = hex;
019        this.complete = complete;
020    }
021
022    public  String getHex() {
023        return hex;
024    }
025
026    public boolean isComplete() {
027        return complete;
028    }
029}
