032/**
033 *
034 */
035public class ReactiveKnowmExchangeProvider implements ReactiveExchange {
036    private static final Logger log = LoggerFactory.getLogger(ReactiveKnowmExchangeProvider.class);
037    protected final String exchangeClassName;
038    protected final Map<CurrencyUnit, String> tickerSymbolConversions;
039    protected final Observable<Long> interval;
040    protected final ExchangeSpecification exchangeSpecification;
041    
042    protected final Map<CurrencyUnitPair, MonitoredCurrency> observablePairs = new HashMap<>();
043
044    Function<CurrencyUnitPair, ExchangeRateUpdate> exchangePollingFunction;
045
046    protected String name;
047    protected ProviderContext providerContext;
048    protected MarketDataService marketDataService;
049
050    private volatile boolean initialized = false;
051    protected Exchange exchange;
052
053    public ReactiveKnowmExchangeProvider(String exchangeClassName,
054                                         Map<CurrencyUnit, String> tickerSymbolConversions,
055                                         Collection<CurrencyUnitPair> pairs,
056                                         Observable<Long> interval) {
057        this(null, exchangeClassName, tickerSymbolConversions, pairs, interval);
058    }
059
060    public ReactiveKnowmExchangeProvider(ExchangeSpecification exchangeSpecification,
061                                         Map<CurrencyUnit, String> tickerSymbolConversions,
062                                         Collection<CurrencyUnitPair> pairs,
063                                         Observable<Long> interval) {
064        this(exchangeSpecification, null, tickerSymbolConversions, pairs, interval);
065    }
066
067    public ReactiveKnowmExchangeProvider(ExchangeSpecification exchangeSpecification,
068                                          String exchangeClassName,
069                                          Map<CurrencyUnit, String> tickerSymbolConversions,
070                                          Collection<CurrencyUnitPair> pairs,
071                                          Observable<Long> interval) {
072        this.exchangeSpecification = exchangeSpecification;
073        if (exchangeSpecification != null) {
074            this.exchangeClassName = exchangeSpecification.getExchangeClassName();
075        } else {
076            this.exchangeClassName = exchangeClassName;
077        }
078        this.tickerSymbolConversions = (tickerSymbolConversions != null) ? tickerSymbolConversions : Collections.emptyMap();
079        this.interval = interval;
080        for (CurrencyUnitPair pair : pairs) {
081            MonitoredCurrency observablePair = new MonitoredCurrency(pair, xchangePair(pair));
082            observablePairs.put(pair, observablePair);
083        }
084    }
085    
086    private synchronized void initialize()  {
087        if (!initialized) {
088            log.info("initializing, calling createExchange()");
089            // createExchange w/o exchangeSpecification does a remote initialization that takes approximately 1.5 seconds
090            if (exchangeSpecification != null) {
091                exchange = ExchangeFactory.INSTANCE.createExchange(exchangeSpecification);
092            } else {
093                exchange = ExchangeFactory.INSTANCE.createExchange(exchangeClassName);
094            }
095            log.info("initializing, getting name");
096            name = exchange.getExchangeSpecification().getExchangeName();
097            providerContext = ProviderContext.of(name, RateType.DEFERRED);
098            marketDataService = exchange.getMarketDataService();
099            initialized = true;
100            log.info("initialized");
101        }
102        log.info("exiting");
103    }
104
105    @Override
106    public String getName() {
107        return name;
108    }
109
110    @Override
111    public PolledObservablePair getObservablePair(CurrencyUnitPair pair) {
112        return observablePairs.get(pair).observable;
113    }
114
115    protected class MonitoredCurrency {
116        final public CurrencyUnitPair  pair;        // Terminating (target) JavaMoney CurrencyUnit
117        final public CurrencyPair exchangePair;     // XChange currency pair (format used by XChange/exchange)
118        final public PolledObservablePair observable;
119        
120        MonitoredCurrency(CurrencyUnitPair pair, CurrencyPair exchangePair) {
121            this.pair = pair;
122            this.exchangePair = exchangePair;
123            observable = new KnowmObservablePair(pair, this::pollingFunction, interval);
124        }
125
126        public ExchangeRateUpdate pollingFunction() {
127            Ticker ticker = pollForTicker();
128            return tickerMapper(ticker);
129        }
130
131        public Ticker pollForTicker() {
132            try {
133                return marketDataService.getTicker(exchangePair);
134            } catch (IOException e) {
135                throw new RuntimeException(e);
136            }
137        }
138
139        public ExchangeRateUpdate tickerMapper(Ticker ticker) {
140            Date timestamp = ticker.getTimestamp();
141            long time = (timestamp != null) ? timestamp.getTime() : 0;
142            return new ExchangeRateUpdate(pair,
143                    DefaultNumberValue.of(ticker.getBid()),
144                    time);
145        }
146    }
147
148    /**
149     * Subclass PolledObservablePair to lazy initialize exchange on subscribe
150     */
151    private class KnowmObservablePair extends PolledObservablePair {
152
153        /**
154         * @param pair         The currency pair to observe
155         * @param rateSupplier A polling Supplier function (for a single pair)
156         * @param interval     An interval (or equivalent) to provide clock ticks to trigger polling
157         */
158        private KnowmObservablePair(CurrencyUnitPair pair, Supplier<ExchangeRateUpdate> rateSupplier, Observable<Long> interval) {
159            super(pair, rateSupplier, interval);
160        }
161
162
163        @Override
164        protected void subscribeActual(@NonNull Observer<? super ExchangeRateUpdate> observer) {
165            initialize();
166            super.subscribeActual(observer);
167        }
168    }
169
170
171    /**
172     * Map from CurrencyUnitPair to XChange CurrencyPair
173     * tickerSymbolConversions is used to handle cases like ItBit that uses "XBT" instead of "BTC"
174     * @param pair  CurrencyUnitPair using JavaMoney CurrencyUnits
175     * @return  XChange CurrencyPair with exchange-specific symbols if any
176     */
177    protected CurrencyPair xchangePair(CurrencyUnitPair pair) {
178        return new CurrencyPair(convertSymbol(pair.getBase()),
179                convertSymbol(pair.getTarget()));
180    }
181
182    /**
183     * Convert a JavaMoney CurrencyUnit to an XChange currency code string
184     * (This will be exchange-specific (e.g. ItBit uses "XBT" instead of "BTC")
185     * @param currencyUnit A JavaMoney currency unit
186     * @return exchange-specific symbol for the currency
187     */
188    public String convertSymbol(CurrencyUnit currencyUnit) {
189        return tickerSymbolConversions.getOrDefault(currencyUnit, currencyUnit.getCurrencyCode());
190    }
