001package org.consensusj.decentralizedid.btcr;
002
003import org.consensusj.decentralizedid.Did;
004
005import java.net.URI;
006import java.net.URISyntaxException;
007
008/**
009 * BTCR DID
010 */
011public class BtcrDid extends Did {
012    public static final String METHOD = "btcr";
013
014    protected BtcrDid(URI uri) {
015        super(uri);
016        String ssp = uri.getSchemeSpecificPart();
017        if (!ssp.startsWith(METHOD + ":")) {
018            fail(uri, "Method is not 'btcr'");
019        }
020    }
021
022    protected BtcrDid(String uriString) throws URISyntaxException {
023        this(new URI(uriString));
024    }
025
026
027    private static void fail(URI candidateUri, String reason) throws IllegalArgumentException {
028        throw new IllegalArgumentException(candidateUri.toString() + " is not a valid DID:BTCR URI (" + reason + ")");
029    }
030
031    /**
032     *
033     * @param uriString A string representing a DID URI
034     * @return a valid DID
035     * @throws  URISyntaxException
036     *          If the URI string constructed from the given components
037     *          violates RFC&nbsp;2396
038     */
039    public static BtcrDid of(String uriString) throws URISyntaxException {
040        return new BtcrDid(uriString);
041    }
042
043    /**
044     *
045     * @param didUri a DID URI
046     * @return a valid DID
047     *
048     */
049    public static BtcrDid of(URI didUri) {
050        return new BtcrDid(didUri);
051    }
052
053    /**
054     * Create a DID from a known good string
055     *
056     * Similar to `java.net.url#create`
057     *
058     * @param uriString
059     * @return a valid DID
060     *
061     * @throws  NullPointerException
062     *          If {@code str} is {@code null}
063     *
064     * @throws  IllegalArgumentException
065     *          If the given string violates RFC&nbsp;2396
066     */
067    public static BtcrDid create(String uriString) {
068        return new BtcrDid(URI.create(uriString));
069    }
070
071}
