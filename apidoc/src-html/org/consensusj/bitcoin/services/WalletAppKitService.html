001package org.consensusj.bitcoin.services;
002
003import com.msgilligan.bitcoinj.json.pojo.ServerInfo;
004import com.msgilligan.bitcoinj.rpcserver.BitcoinJsonRpc;
005import org.bitcoinj.core.Coin;
006import org.bitcoinj.core.Context;
007import org.bitcoinj.core.NetworkParameters;
008import org.bitcoinj.core.PeerGroup;
009import org.bitcoinj.core.Transaction;
010import org.bitcoinj.kits.WalletAppKit;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import javax.annotation.PostConstruct;
015import javax.inject.Inject;
016import javax.inject.Named;
017import java.math.BigDecimal;
018
019/**
020 * Implement a subset of Bitcoin JSON RPC using a WalletAppKit
021 */
022@Named
023public class WalletAppKitService implements BitcoinJsonRpc {
024    private static Logger log = LoggerFactory.getLogger(WalletAppKitService.class);
025    private static final String userAgentName = "PeerList";
026    private static final String appVersion = "0.1";
027    private static final int version = 1;
028    private static final int protocolVersion = 1;
029    private static final int walletVersion = 0;
030
031    protected NetworkParameters netParams;
032    protected Context context;
033    protected WalletAppKit kit;
034
035    private int timeOffset = 0;
036    private BigDecimal difficulty = new BigDecimal(0);
037
038    @Inject
039    public WalletAppKitService(NetworkParameters params,
040                               Context context,
041                               WalletAppKit kit) {
042        this.netParams = params;
043        this.context = context;
044        this.kit = kit;
045    }
046
047    @PostConstruct
048    public void start() {
049        log.info("WalletAppKitService.start()");
050        kit.setUserAgent(userAgentName, appVersion);
051        kit.setBlockingStartup(false);
052        kit.startAsync();
053        //kit.awaitRunning();
054    }
055
056    public NetworkParameters getNetworkParameters() {
057        return this.netParams;
058    }
059
060    public PeerGroup getPeerGroup() {
061        kit.awaitRunning();
062        return kit.peerGroup();
063    }
064
065    @Override
066    public Integer getblockcount() {
067        log.info("getblockcount");
068        if(!kit.isRunning()) {
069            log.warn("kit not running, returning null");
070            return null;
071        }
072        return kit.chain().getChainHead().getHeight();
073    }
074
075    @Override
076    public Integer getconnectioncount() {
077        if(!kit.isRunning()) {
078            return null;
079        }
080        try {
081            return kit.peerGroup().numConnectedPeers();
082        } catch (IllegalStateException ex) {
083            return null;
084        }
085    }
086
087    @Override
088    public ServerInfo getinfo() {
089        // Dummy up a response for now.
090        // Since ServerInfo is immutable, we have to build it entirely with the constructor.
091        Coin balance = Coin.valueOf(0);
092        boolean testNet = !netParams.getId().equals(NetworkParameters.ID_MAINNET);
093        int keyPoolOldest = 0;
094        int keyPoolSize = 0;
095        return new ServerInfo(
096                version,
097                protocolVersion,
098                walletVersion,
099                balance,
100                getblockcount(),
101                timeOffset,
102                getconnectioncount(),
103                "proxy",
104                difficulty,
105                testNet,
106                keyPoolOldest,
107                keyPoolSize,
108                Transaction.REFERENCE_DEFAULT_MIN_TX_FEE,
109                Transaction.REFERENCE_DEFAULT_MIN_TX_FEE, // relayfee
110                "no errors"                               // errors
111        );
112    }
113
114}
