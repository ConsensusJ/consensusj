001package org.consensusj.bitcoin.cli;
002
003import com.msgilligan.bitcoinj.rpc.BitcoinClient;
004import com.msgilligan.bitcoinj.rpc.RpcConfig;
005import com.msgilligan.bitcoinj.rpc.bitcoind.BitcoinConfFile;
006import org.bitcoinj.core.NetworkParameters;
007import org.bitcoinj.params.MainNetParams;
008import org.bitcoinj.params.RegTestParams;
009import org.bitcoinj.params.TestNet3Params;
010import org.consensusj.jsonrpc.JsonRpcException;
011import org.consensusj.jsonrpc.cli.CliCommand;
012
013import java.net.URI;
014import java.net.URISyntaxException;
015
016/**
017 *
018 */
019public abstract class BitcoinCliCommand extends CliCommand {
020    protected BitcoinClient client = null;
021
022    protected BitcoinCliCommand(String name, CliOptions options, String[] args) {
023        this(name, name, options, args);
024    }
025
026    protected BitcoinCliCommand(String name, String usage, CliOptions options, String[] args) {
027        super(name, usage, options, args);
028    }
029
030    public BitcoinClient getClient() {
031        if (client == null) {
032            System.out.println("Connecting to: " + getRPCConfig().getURI());
033            RpcConfig config = getRPCConfig();
034            client = new BitcoinClient( config.getNetParams(),
035                    config.getURI(),
036                    config.getUsername(),
037                    config.getPassword());
038        }
039        return client;
040    }
041
042    /**
043     * Initial a client and if rpcwait option set, make sure server is accepting connections.
044     * @return status code
045     */
046    public int preflight() {
047        getClient();
048        if (line.hasOption("rpcwait")) {
049            boolean available = false;   // Wait up to 1 hour
050            try {
051                available = client.waitForServer(60*60);
052            } catch (JsonRpcException e) {
053                this.pwerr.println("JSON-RPC Exception: " + e.getMessage());
054                return 1;
055            }
056            if (!available) {
057                this.pwerr.println("Timeout error.");
058                return 1;
059            }
060        }
061        return 0;
062    }
063
064    protected URI getServerURI(URI confFileURI) {
065        String scheme = confFileURI.getScheme();
066        String host = confFileURI.getHost();
067        int port = confFileURI.getPort();
068        String file = confFileURI.getPath();
069
070        if (line.hasOption("rpcssl")) {
071            scheme = "https";
072        }
073        if (line.hasOption("rpcconnect")) {
074            host = line.getOptionValue("rpcconnect", host);
075        }
076        if (line.hasOption("regtest") || line.hasOption("testnet")) {
077            port = 18332;
078        }
079        if (line.hasOption("rpcport")) {
080            String portString = line.getOptionValue("rpcport");
081            port = Integer.parseInt(portString);
082        }
083        URI rpcServerURI = null;
084        try {
085            rpcServerURI = new URI(scheme, null, host, port, file, null, null);
086        } catch (URISyntaxException e) {
087            // We should be careful that this never happens
088            e.printStackTrace();
089            // But if it does, throw an unchecked exception
090            throw new RuntimeException(e);
091        }
092        return rpcServerURI;
093    }
094
095    protected RpcConfig getRPCConfig() {
096        RpcConfig confFileConfig = BitcoinConfFile.readDefaultConfig().getRPCConfig();
097        URI uri = getServerURI(confFileConfig.getURI());
098        String user = line.getOptionValue("rpcuser", confFileConfig.getUsername());
099        String pass = line.getOptionValue("rpcpassword", confFileConfig.getPassword());
100        NetworkParameters netParams;
101        if (line.hasOption("regtest")) {
102            netParams = RegTestParams.get();
103        } else if (line.hasOption(("testnet"))) {
104            netParams = TestNet3Params.get();
105        } else {
106            netParams = MainNetParams.get();
107        }
108        RpcConfig cfg = new RpcConfig(netParams, uri, user, pass);
109        return cfg;
110    }
111
112}
