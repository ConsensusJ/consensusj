001package org.consensusj.jsonrpc.introspection;
002
003import java.lang.reflect.Method;
004import java.util.Map;
005
006/**
007 * Implementation of JsonRpcServerWrapper that takes a delegate
008 * object in the constructor
009 */
010public class DelegatingJsonRpcService extends AbstractJsonRpcService {
011    private final Object service;
012
013    /**
014     * Use this constructor for simplicity and if you don't need GraalVM support
015     * @param serviceObject
016     */
017    public DelegatingJsonRpcService(Object serviceObject) {
018        this(JsonRpcServiceWrapper.reflect(serviceObject.getClass()), serviceObject);
019    }
020
021    /**
022     * Use this constructor for GraalVM compatibility (methods generated in static initializer)
023     *
024     * @param serviceObject
025     * @param methods
026     */
027    public DelegatingJsonRpcService(Map<String, Method> methods, Object serviceObject) {
028        super(methods);
029        this.service = serviceObject;
030    }
031
032    @Override
033    public Object getServiceObject() {
034        return service;
035    }
036
037    @Override
038    public Method getMethod(String methodName) {
039        return methods.get(methodName);
040    }
041}
