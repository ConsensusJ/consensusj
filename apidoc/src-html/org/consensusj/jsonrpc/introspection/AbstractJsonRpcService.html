001package org.consensusj.jsonrpc.introspection;
002
003import java.lang.reflect.Method;
004import java.util.Map;
005
006/**
007 * Graal-compatible implementation that takes a map of Methods in the constructor
008 * so that introspection can be done at static initialization time. Assumes
009 * implementation is in a subclass.
010 * (which means during native image generation)
011 */
012public abstract class AbstractJsonRpcService implements JsonRpcServiceWrapper {
013    protected Map<String, Method> methods;
014
015    public AbstractJsonRpcService(Map<String, Method> methods) {
016        this.methods = methods;
017    }
018
019    @Override
020    public Object getServiceObject() {
021        return this;
022    }
023
024    @Override
025    public Method getMethod(String methodName) {
026        return methods.get(methodName);
027    }
028
029}
